SQL Banking Analytics – KPI Calculation

Below is the SQL logic used to calculate banking indicators such as number of accounts, transaction metrics, and aggregated KPIs using temporary tables.

Step 1 – Calculating the total number of account owned by client


/*
. Numero totale di conti posseduti.
. Numero di conti posseduti per tipologia (un indicatore per ogni tipo di conto).
*/
DROP TABLE IF EXISTS banca.temp_indicatori_conti;
CREATE TEMPORARY TABLE banca.temp_indicatori_conti(
SELECT 
conto.id_cliente,
COUNT(conto.id_conto) AS numero_conti,
SUM( CASE WHEN tc.desc_tipo_conto = 'Conto Base' THEN 1 ELSE 0 END) AS conto_base,
SUM( CASE WHEN tc.desc_tipo_conto = 'Conto Business' THEN 1 ELSE 0 END) AS conto_business,
SUM( CASE WHEN tc.desc_tipo_conto = 'Conto Privati' THEN 1 ELSE 0 END) AS conto_privati,
SUM( CASE WHEN tc.desc_tipo_conto = 'Conto Famiglie' THEN 1 ELSE 0 END) AS conto_famiglie
FROM banca.conto INNER JOIN banca.tipo_conto tc ON conto.id_tipo_conto = tc.id_tipo_conto
GROUP BY 1
ORDER BY 2 DESC
);

Step 2 – Calculating transactions


/*
. Numero di transazioni in uscita su tutti i conti.
. Numero di transazioni in entrata su tutti i conti.
. Importo totale transato in uscita su tutti i conti.
. Importo totale transato in entrata su tutti i conti.
. Numero di transazioni in uscita per tipologia di conto (un indicatore per tipo di conto).
. Numero di transazioni in entrata per tipologia di conto (un indicatore per tipo di conto).
. Importo transato in uscita per tipologia di conto (un indicatore per tipo di conto).
. Importo transato in entrata per tipologia di conto (un indicatore per tipo di conto).
*/

DROP TABLE IF EXISTS banca.temp_indicatori_transazioni;
CREATE TEMPORARY TABLE banca.temp_indicatori_transazioni(
SELECT 
conto.id_cliente,
SUM(CASE WHEN tt.segno = '+' THEN 1 ELSE 0 END) AS num_incoming,
SUM(CASE WHEN tt.segno = '-' THEN 1 ELSE 0 END) AS num_outgoing,
ROUND(SUM(CASE WHEN tt.segno = '+' THEN transazioni.importo ELSE 0 END), 2) AS tot_incoming,
ROUND(SUM(CASE WHEN tt.segno = '-' THEN transazioni.importo ELSE 0 END),2) AS tot_outgoing,
SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Base' THEN 1 ELSE 0 END) AS num_incoming_c_base,
SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Business' THEN 1 ELSE 0 END) AS num_incoming_c_business,
SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Privati' THEN 1 ELSE 0 END) AS num_incoming_c_privati,
SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Famiglie' THEN 1 ELSE 0 END) AS num_incoming_c_famiglie,
SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Base' THEN 1 ELSE 0 END) AS num_outgoing_c_base,
SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Business' THEN 1 ELSE 0 END) AS num_outgoing_c_business,
SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Privati' THEN 1 ELSE 0 END) AS num_outgoing_c_privati,
SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Famiglie' THEN 1 ELSE 0 END) AS num_outgoing_c_famiglie,
ROUND(SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Base' THEN transazioni.importo ELSE 0 END), 2) AS tot_incoming_c_base,
ROUND(SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Business' THEN transazioni.importo ELSE 0 END), 2) AS tot_incoming_c_business,
ROUND(SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Privati' THEN transazioni.importo ELSE 0 END), 2) AS tot_incoming_c_privati,
ROUND(SUM(CASE WHEN tt.segno = '+' AND tc.desc_tipo_conto = 'Conto Famiglie' THEN transazioni.importo ELSE 0 END), 2) AS tot_incoming_c_famiglie,
ROUND(SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Base' THEN transazioni.importo ELSE 0 END),2) AS tot_outgoing_c_base,
ROUND(SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Business' THEN transazioni.importo ELSE 0 END),2) AS tot_outgoing_c_business,
ROUND(SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Privati' THEN transazioni.importo ELSE 0 END),2) AS tot_outgoing_c_privati,
ROUND(SUM(CASE WHEN tt.segno = '-' AND tc.desc_tipo_conto = 'Conto Famiglie' THEN transazioni.importo ELSE 0 END),2) AS tot_outgoing_c_famiglie
FROM banca.transazioni 
INNER JOIN banca.conto ON transazioni.id_conto = conto.id_conto
INNER JOIN banca.tipo_conto tc ON conto.id_tipo_conto = tc.id_tipo_conto
INNER JOIN banca.tipo_transazione tt ON transazioni.id_tipo_trans = tt.id_tipo_transazione
GROUP BY 1
);

Step 3 – Final query joining with temporary tables


SELECT 
cliente.*,
YEAR(CURRENT_DATE) - YEAR(cliente.data_nascita) AS age, -- calculating age on client table
TIMESTAMPDIFF(YEAR, cliente.data_nascita, current_date()) AS age2, -- usando TIMESTAMPDIFF
ROUND(DATEDIFF(CURRENT_DATE(), cliente.data_nascita) / 365, 0) as age3, -- usando DATEDIFF
tic.*,
tit.*
FROM banca.cliente LEFT JOIN banca.temp_indicatori_conti tic ON cliente.id_cliente = tic.id_cliente
LEFT JOIN banca.temp_indicatori_transazioni tit ON cliente.id_cliente = tit.id_cliente
