MongoDB – Contact List Management

Below is the code used for querying and updating the MongoDB contact list.

Queries


// Find all contacts associated with WebCorp
db.ContactList.find({ Società: "WebCorp" })

// Identify contacts with more than one phone number
db.ContactList.find({ "Numero_di_cellulare.1": { $exists: true } })

// Extract phone numbers of contacts tagged as "lavoro"
db.ContactList.find(
  { Tag: "lavoro" },
  { Nome: 1, Cognome: 1, Numero_di_cellulare: 1 }
)

Aggregations


// Count contacts marked as "amici stretti"
db.ContactList.aggregate([
  { $group: { _id: { Close_friends: "$Amici_stretti" }, count: { $sum: 1 } } }
])

// Average calls last month for close friends
db.ContactList.aggregate([
  { $match: { Amici_stretti: true } },
  { $group: { _id: "Close Friends", avg_call_last_month: { $avg: "$Chiamate_ultimo_mese" } } }
])

Updates


// Add phone number to Simone Azzurri
db.ContactList.updateOne(
  { Nome: "Simone", Cognome: "Azzurri" },
  { $push: { Numero_di_cellulare: "345678902" } }
)

// Insert new contact
db.ContactList.insertOne({
  Nome: "Mary",
  Cognome: "Salgado",
  Numero_di_cellulare: "346679933",
  Altri_contatti: { Indirizzo: "Via 25 Aprile 3, Firenze" }
})
